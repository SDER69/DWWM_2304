VARIABLES
	
	a est un entier.
	b est un entier.
	c est un entier.

TRAITEMENT

	LIRE a.

	ECRIRE "Veuillez entrer un second nombre entier b : "
	LIRE b.

	ECRIRE "Veuillez entrer un premier nombre entier c : "
	LIRE c.
